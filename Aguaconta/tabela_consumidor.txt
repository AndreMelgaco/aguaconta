	Tipo Instalação (Residencial / Comercial)
	Telefone
	
	
CREATE TABLE CONSUMIDOR
(
	HIDROMETRO INT NOT NULL PRIMARY KEY,
	NOME VARCHAR(100) NOT NULL,
	CPF VARCHAR(11) NOT NULL,
	UNID_CONSUMO VARCHAR(MAX) NOT NULL,
	ENDERECO VARCHAR(MAX) NOT NULL,
	TIPO_INST CHAR NOT NULL DEFAULT 'R',
	TEL_FIXO VARCHAR(MAX),
	TEL_CELULAR VARCHAR(MAX),
	DT_INICIO DATETIME NOT NULL DEFAULT getdate(),
	DT_CADASTRO DATETIME NOT NULL DEFAULT getdate()
)
